CREATE TABLE `tb_board_article` (
	`bda_seq` BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '게시글 번호',
	`bd_seq` INT(11) NOT NULL DEFAULT '0' COMMENT '게시판 번호',
	`bda_title` VARCHAR(2000) NULL DEFAULT NULL COMMENT '게시글 제목' COLLATE 'utf8mb4_general_ci',
	`bda_content` LONGTEXT NULL DEFAULT NULL COMMENT '게시글 내용' COLLATE 'utf8mb4_general_ci',
	`bda_view_cnt` INT(11) NULL DEFAULT '0' COMMENT '게시글 조회수',
	`regdate` DATETIME NULL DEFAULT NULL COMMENT '등록일시',
	`reguser` VARCHAR(100) NULL DEFAULT NULL COMMENT '등록자' COLLATE 'utf8mb4_general_ci',
	`moddate` DATETIME NULL DEFAULT NULL COMMENT '수정일시',
	`moduser` VARCHAR(100) NULL DEFAULT NULL COMMENT '수정자' COLLATE 'utf8mb4_general_ci',
	PRIMARY KEY (`bda_seq`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=8
;
